# Re-cargar datos (copiar el bloque anterior si es necesario)

# ---- 1. Análisis descriptivo ----

<RCell id="exA-datos" title="Ejercicio A: Simular y explorar datos de crecimiento de plantas" code={`set.seed(2024)

n_plantas <- 10   # por grupo
n_semanas <- 8
grupos <- c("Control", "FertA", "FertB")

# Efectos fijos: intercepto, pendiente por grupo
beta0 <- 5           # altura inicial promedio
beta_ctrl <- 2.0     # pendiente Control
beta_A    <- 3.0     # pendiente FertA (mayor crecimiento)
beta_B    <- 2.5     # pendiente FertB (crecimiento intermedio)

sigma_b1 <- 0.3  # varianza de la pendiente aleatoria
sigma_e  <- 0.8  # varianza residual

  pendiente <- c(beta_ctrl, beta_A, beta_B)[g]

    id_planta <- (g - 1) * n_plantas + p

    b1_i <- rnorm(1, 0, sigma_b1)

              rnorm(n_semanas, 0, sigma_e)

               semana = semana, altura = altura)

}))

### Enunciado

Simula un ensayo clínico con 40 sujetos (20 por grupo), 6 visitas, y presencia de datos faltantes (dropout aleatorio). Ajusta un LMM con estructura CS y evalúa si el patrón de datos faltantes afecta las conclusiones (**análisis de sensibilidad**).

<RCell id="exB-simulacion" title="Ejercicio B: Simular ensayo clínico con dropout" code={`set.seed(42)
library(lme4)
library(ggplot2)

n_suj <- 20  # por grupo
n_vis <- 6

# Simular datos completos
sim_ensayo <- do.call(rbind, lapply(1:(2 * n_suj), function(i) {
  grupo <- ifelse(i <= n_suj, "Placebo", "Activo")
  pendiente <- ifelse(grupo == "Activo", -2.5, -0.5)  # el activo mejora más
  b0 <- rnorm(1, 0, 5)  # intercepto aleatorio
  b1 <- rnorm(1, 0, 0.5)  # pendiente aleatoria

    sujeto  = factor(i),

    grupo   = grupo,

dropout_mask[sim_ensayo$visita == 0] <- FALSE  # baseline siempre presente
sim_dropout <- sim_ensayo
sim_dropout$respuesta[dropout_mask] <- NA
